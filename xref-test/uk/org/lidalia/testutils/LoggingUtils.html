25  package uk.org.lidalia.testutils;
26  
27  import org.powermock.reflect.Whitebox;
28  import org.slf4j.LoggerFactory;
29  
30  import uk.org.lidalia.sysoutslf4j.common.ExceptionUtils;
31  
32  import ch.qos.logback.classic.Level;
33  import ch.qos.logback.classic.Logger;
34  import ch.qos.logback.classic.LoggerContext;
35  
36  public class LoggingUtils {
37  
38  	public static void turnOffRootLogging() {
39  		LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
40  		Logger rootLogger = lc.getLogger(Logger.ROOT_LOGGER_NAME);
41  		rootLogger.setLevel(Level.OFF);
42  		rootLogger.detachAndStopAllAppenders();
43  	}
44  
45  	public static void turnOffRootLogging(ClassLoader classLoader) {
46  		try {
47  			Class<?> loggingUtils = classLoader.loadClass(LoggingUtils.class.getName());
48  			Whitebox.invokeMethod(loggingUtils, "turnOffRootLogging");
49  		} catch (Exception e) {
50  			throw ExceptionUtils.asRuntimeException(e);
51  		}
52  	}
53  	
54  	private LoggingUtils() {
55  		throw new UnsupportedOperationException("Not instantiable");
56  	}
57  }
