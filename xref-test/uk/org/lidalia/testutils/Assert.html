43  	public static <T extends Throwable> void shouldThrow(final T expectedThrowable, Callable<Void> workThatShouldThrowThrowable) throws Throwable {
44  		T actualThrowable = shouldThrow(getClass(expectedThrowable), workThatShouldThrowThrowable);
45  		assertSame(expectedThrowable, actualThrowable);
46  	}
47  
48  	@SuppressWarnings("unchecked")
49  	public static <T extends Throwable> T shouldThrow(Class<T> expectedThrowableType, Callable<Void> workThatShouldThrowThrowable) throws Throwable {
50  		try {
51  			workThatShouldThrowThrowable.call();
52  		} catch (Throwable actualThrowableThrown) {
53  			if (instanceOf(actualThrowableThrown, expectedThrowableType)) {
54  				return (T) actualThrowableThrown;
55  			} else {
56  				throw actualThrowableThrown;
57  			}
58  		}
59  		throw new AssertionError("No exception thrown");
60  	}
61  
62  	@SuppressWarnings("unchecked")
63  	private static <T> Class<? extends T> getClass(final T object) {
64  		return (Class<? extends T>) object.getClass();
65  	}

67  	public static boolean instanceOf(Object o, Class<?> c) {
68  		return c.isAssignableFrom(o.getClass());
69  	}

90  	public static void assertExpectedLoggingEvent(
91  			ILoggingEvent loggingEvent, String message, Level level, Marker marker, String className, Throwable throwable) {
92  		assertEquals(message, loggingEvent.getMessage());
93  		assertEquals(level, loggingEvent.getLevel());
94  		assertEquals(className, loggingEvent.getLoggerName());
95  		assertEquals(marker, loggingEvent.getMarker());
96  		IThrowableProxy throwableProxy = loggingEvent.getThrowableProxy();
97  		assertThrowableProxyEquals(throwable, throwableProxy);
98  	}

100 	private static void assertThrowableProxyEquals(Throwable throwable, IThrowableProxy throwableProxy) {
101 		if (throwable != null) {
102 			assertEquals(throwable.getClass().getName(), throwableProxy.getClassName());
103 			assertEquals(throwable.getMessage(), throwableProxy.getMessage());
104 			assertThrowableProxyEquals(throwable.getCause(), throwableProxy.getCause());
105 		} else {
106 			assertNull(throwableProxy);
107 		}
108 	}

110 	public static void assertNotInstantiable(final Class<?> classThatShouldNotBeInstantiable) throws Throwable {
111 		assertOnlyHasNoArgsConstructor(classThatShouldNotBeInstantiable);
112 
113 		UnsupportedOperationException oue = shouldThrow(UnsupportedOperationException.class, new Callable<Void>() {
114 			public Void call() throws Exception {
115 				Whitebox.invokeConstructor(classThatShouldNotBeInstantiable);
116 				return null;
117 			}
118 		});
119 		assertEquals("Not instantiable", oue.getMessage());
120 	}
121 
122 	private static void assertOnlyHasNoArgsConstructor(final Class<?> classThatShouldNotBeInstantiable) {
123 		assertEquals(Object.class, classThatShouldNotBeInstantiable.getSuperclass());
124 		assertEquals(1, classThatShouldNotBeInstantiable.getDeclaredConstructors().length);
125 		final Constructor<?> constructor = classThatShouldNotBeInstantiable.getDeclaredConstructors()[0];
126 		assertEquals(0, constructor.getParameterTypes().length);
127 	}
