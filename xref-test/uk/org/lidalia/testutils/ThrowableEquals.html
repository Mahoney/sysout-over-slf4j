25  package uk.org.lidalia.testutils;
26  
27  import org.easymock.EasyMock;
28  import org.easymock.IArgumentMatcher;
29  
30  public class ThrowableEquals implements IArgumentMatcher {
31  	
32  	private Throwable expected;
33  
34      public ThrowableEquals(Throwable expected) {
35          this.expected = expected;
36      }
37  
38  
39  	public void appendTo(StringBuffer buffer) {
40  		buffer.append("eqExceptionCause(");
41          buffer.append(expected.toString());
42          buffer.append(" with cause \"");
43          buffer.append(expected.getCause());
44          buffer.append("\")");
45  
46  	}
47  
48  	public boolean matches(Object actual) {
49  		if (!(actual instanceof Throwable)) {
50              return false;
51          }
52          Throwable actualThrowable = (Throwable) actual;
53          return expected == actualThrowable.getCause();
54  	}
55  	
56  	public static <T extends Throwable> T eqExceptionCause(T in) {
57  	    EasyMock.reportMatcher(new ThrowableEquals(in));
58  	    return null;
59  	}
60  
61  
62  }
