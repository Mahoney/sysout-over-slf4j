25  package uk.org.lidalia.sysoutslf4j.common;
26  
27  import java.lang.reflect.InvocationHandler;
28  import java.lang.reflect.Method;
29  import java.util.HashMap;
30  import java.util.Map;
31  
32  public class ProxyingInvocationHandler implements InvocationHandler {
33  
34  	private final Object target;
35  	private final Map<Method, Method> methods = new HashMap<Method, Method>();
36  	
37  	ProxyingInvocationHandler(final Object target, final Class<?> interfaceClass) {
38  		this.target = target;
39  		for (Method method : interfaceClass.getMethods()) {
40  			try {
41  				final Method methodOnTarget = target.getClass().getMethod(method.getName(), method.getParameterTypes());
42  				methods.put(method, methodOnTarget);
43  			} catch (NoSuchMethodException e) {
44  				throw new IllegalArgumentException(
45  						"Target " + target + " does not have methods to match all method signatures on class " + interfaceClass,
46  						e);
47  			}
48  		}
49  	}
50  
51  	@Override
52  	public Object invoke(final Object proxy, final Method method, final Object[] args) throws Throwable {
53  		final Method methodOnTarget = methods.get(method);
54  		return methodOnTarget.invoke(target, args);
55  	}
56  	
57  	public Object getTarget() {
58  		return target;
59  	}
60  }
