25  package uk.org.lidalia.sysoutslf4j.system;
26  
27  import java.io.PrintStream;
28  
29  import uk.org.lidalia.sysoutslf4j.common.PrintStreamCoordinator;
30  import uk.org.lidalia.sysoutslf4j.common.SLF4JPrintStream;
31  import uk.org.lidalia.sysoutslf4j.common.SystemOutput;
32  
33  public final class PrintStreamCoordinatorImpl implements PrintStreamCoordinator {
34  	
35  	@Override
36  	public void replaceSystemOutputsWithSLF4JPrintStreams() {
37  		for (SystemOutput systemOutput : SystemOutput.values()) {
38  			replaceSystemOutputWithSLF4JPrintStream(systemOutput);
39  		}
40  	}
41  
42  	private static void replaceSystemOutputWithSLF4JPrintStream(final SystemOutput systemOutput) {
43  		final SLF4JPrintStreamImpl slf4jPrintStream = buildSLF4JPrintStream(systemOutput.get());
44  		systemOutput.set(slf4jPrintStream);
45  	}
46  
47  	private static SLF4JPrintStreamImpl buildSLF4JPrintStream(final PrintStream originalPrintStream) {
48  		final LoggerAppenderStore loggerAppenderStore = new LoggerAppenderStore();
49  		final SLF4JPrintStreamDelegate delegate = new SLF4JPrintStreamDelegate(originalPrintStream, loggerAppenderStore);
50  		return new SLF4JPrintStreamImpl(originalPrintStream, delegate);
51  	}
52  
53  	@Override
54  	public void restoreOriginalSystemOutputs() {
55  		for (SystemOutput systemOutput : SystemOutput.values()) {
56  			restoreSystemOutput(systemOutput);
57  		}
58  	}
59  
60  	private static void restoreSystemOutput(final SystemOutput systemOutput) {
61  		final SLF4JPrintStream slf4jPrintStream = (SLF4JPrintStream) systemOutput.get();
62  		systemOutput.set(slf4jPrintStream.getOriginalPrintStream());
63  	}
64  }
